import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'; const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', }; type DiscordMember = { user?: { id?: string; username?: string; global_name?: string | null; }; roles?: string[]; }; Deno.serve(async (req) => { if (req.method === 'OPTIONS') { return new Response('ok', { headers: corsHeaders }); } const supabaseUrl = Deno.env.get('SUPABASE_URL'); const serviceRoleKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY'); const discordBotToken = Deno.env.get('DISCORD_BOT_TOKEN'); const discordGuildId = Deno.env.get('DISCORD_GUILD_ID'); const requiredRoleId = Deno.env.get('DISCORD_REQUIRED_ROLE_ID') || ''; if (!supabaseUrl || !serviceRoleKey || !discordBotToken || !discordGuildId) { return json({ allowed: false, message: 'Discord gate is missing function secrets.' }); } const authHeader = req.headers.get('Authorization') || ''; const jwt = authHeader.replace('Bearer ', '').trim(); if (!jwt) { return json({ allowed: false, message: 'Missing user session.' }); } const supabase = createClient(supabaseUrl, serviceRoleKey); const { data: userData, error: userError } = await supabase.auth.getUser(jwt); if (userError || !userData.user) { return json({ allowed: false, message: 'Invalid user session.' }); } const user = userData.user; const discordIdentity = user.identities?.find((identity) => identity.provider === 'discord'); const identityData = discordIdentity?.identity_data || {}; const discordUserId = String(identityData.provider_id || identityData.sub || identityData.id || ''); if (!discordUserId) { await markDenied(supabase, user.id); return json({ allowed: false, message: 'This account is not connected to Discord.' }); } const memberResponse = await fetch( `https://discord.com/api/v10/guilds/${discordGuildId}/members/${discordUserId}`, { headers: { Authorization: `Bot ${discordBotToken}`, }, }, ); if (memberResponse.status === 404) { await markDenied(supabase, user.id, discordUserId); return json({ allowed: false, message: 'Join the required Discord server before using the map.' }); } if (!memberResponse.ok) { const details = await memberResponse.text(); console.error('Discord member lookup failed:', memberResponse.status, details); let message = `Discord server membership check failed (${memberResponse.status}).`; if (memberResponse.status === 401) { message = 'Discord bot token is invalid. Reset the bot token and update DISCORD_BOT_TOKEN.'; } else if (memberResponse.status === 403) { message = 'Discord bot cannot read that server. Make sure the bot is in the server and has permission to view members.'; } return json({ allowed: false, message, discord_status: memberResponse.status, }); } const member = await memberResponse.json() as DiscordMember; const roles = member.roles || []; const allowed = !requiredRoleId || roles.includes(requiredRoleId); await supabase.from('profiles').upsert({ id: user.id, discord_user_id: discordUserId, discord_username: member.user?.username || String(identityData.user_name || identityData.preferred_username || ''), discord_global_name: member.user?.global_name || String(identityData.full_name || identityData.name || ''), discord_access: allowed, discord_roles: roles, discord_checked_at: new Date().toISOString(), }); if (!allowed) { return json({ allowed: false, message: 'You need the required Discord role before using the map.' }); } return json({ allowed: true, discord_user_id: discordUserId, username: member.user?.username || identityData.user_name || '', global_name: member.user?.global_name || identityData.full_name || '', }); }); async function markDenied(supabase: ReturnType, userId: string, discordUserId = '') { await supabase.from('profiles').upsert({ id: userId, discord_user_id: discordUserId || null, discord_access: false, discord_checked_at: new Date().toISOString(), }); } function json(body: unknown, status = 200) { return new Response(JSON.stringify(body), { status, headers: { ...corsHeaders, 'Content-Type': 'application/json', }, }); }
